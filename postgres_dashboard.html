📜 Common SQL Queries

Suppliers by Country

SELECT country, COUNT(*) as supplier_count FROM suppliers GROUP BY country ORDER BY supplier_count DESC LIMIT 10;

Check AI Enhancement Progress

SELECT COUNT(*) as total, COUNT(CASE WHEN products IS NOT NULL AND LENGTH(products) > 200 THEN 1 END) as enhanced, ROUND(COUNT(CASE WHEN products IS NOT NULL AND LENGTH(products) > 200 THEN 1 END) * 100.0 / COUNT(*), 2) as percentage FROM suppliers;

Search Suppliers by Product

SELECT supplier_name, country, products FROM suppliers WHERE LOWER(products) LIKE '%olive oil%' LIMIT 20;
